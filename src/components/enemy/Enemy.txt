import './enemy.css';
import { useEffect } from "react";

function Enemy ({ run, main, random, enemyTools, setMainEnemy, enemyScore }) {

    useEffect(() => {
            setMainEnemy(enemyTools[random])
    }, [main])



    return (
        <div className="enemy">
            <h2>Score : {enemyScore}</h2>
            <img src={enemyTools[random]} />
            <div className="tools-enemy">
                <h1>Computer</h1>
            </div>
        </div>
    )
}

export default Enemy
